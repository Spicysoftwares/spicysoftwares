-- Example 1: Third-Person Character Controller
function OnUpdate(dt)
    local speed = 5.0
    local x, y, z = GetPosition()
    
    -- Movement
    if IsKeyDown("W") then z = z - speed * dt end
    if IsKeyDown("S") then z = z + speed * dt end
    if IsKeyDown("A") then x = x - speed * dt end
    if IsKeyDown("D") then x = x + speed * dt end
    
    SetPosition(x, y, z)
    
    -- Jump with physics
    if IsKeyDown("SPACE") and IsGrounded() then
        AddForce(0, 8, 0)
        PlayAnim("Jump")
        PlayAudio("assets/sounds/jump.wav")
    end
end
-- Example 2: Enemy AI with Patrol & Chase
local state = "patrol"
local patrolPoints = {
    {0, 0, 0}, {10, 0, 0}, {10, 0, 10}, {0, 0, 10}
}
local currentPoint = 1

function OnUpdate(dt)
    local x, y, z = GetPosition()
    local px, py, pz = GetCameraPosition() -- Player position
    
    local distToPlayer = GetDistance(x, y, z, px, py, pz)
    
    if distToPlayer < 5.0 then
        -- Chase player
        state = "chase"
        LookAt(px, py, pz)
        Translate(0, 0, -3.0 * dt)
        SetColor(1, 0, 0) -- Red when chasing
        PlayAnim("Run")
    else
        -- Patrol mode
        state = "patrol"
        local target = patrolPoints[currentPoint]
        local dist = GetDistance(x, y, z, target[1], target[2], target[3])
        
        if dist < 1.0 then
            currentPoint = currentPoint + 1
            if currentPoint > #patrolPoints then currentPoint = 1 end
        end
        
        LookAt(target[1], target[2], target[3])
        Translate(0, 0, -2.0 * dt)
        SetColor(0, 1, 0) -- Green when patrolling
    end
end
-- Example 3: Floating Platform with Sin Wave
local startY = 2.0
StartTimer("platform")

function OnUpdate(dt)
    local time = GetTimer("platform")
    local x, y, z = GetPosition()
    
    -- Oscillate vertically using sine
    local newY = startY + Sin(time * 2.0) * 2.0
    SetPosition(x, newY, z)
    
    -- Rotate slowly
    local rx, ry, rz = GetRotation()
    SetRotation(rx, ry + 30 * dt, rz)
end
-- Example 4: Interactive Door System
local isOpen = false
local openPos = 5.0
local closedPos = 0.0

function OnUpdate(dt)
    local playerID = FindEntity("Player")
    if playerID == nil then return end
    
    local px, py, pz = GetEntityPosition(playerID)
    local x, y, z = GetPosition()
    
    local dist = GetDistance(x, y, z, px, py, pz)
    
    if dist < 3.0 and IsKeyDown("E") then
        isOpen = not isOpen
        PlayAudio("assets/sounds/door.wav")
    end
    
    -- Smooth lerp to target position
    local target = isOpen and openPos or closedPos
    local newY = Lerp(y, target, 5.0 * dt)
    SetPosition(x, newY, z)
end
